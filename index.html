import streamlit as st import pandas as pd # إعدادات الصفحة st.set_page_config(page_title="مدير ميزانية 2026", layout="centered") st.title("💰 نظام إدارة الميزانية الشخصية 2026") st.write("من 27 / 1 / 2026 حتى 27 / 12 / 2026") # --- القسم الأول: الدخل --- st.header("📥 الدخل الشهري") col1, col2, col3 = st.columns(3) with col1: salary = st.number_input("الراتب الأساسي", value=18072) with col2: citizen_acc = st.number_input("حساب المواطن", value=1100) with col3: support = st.number_input("الدعم", value=1913) total_income = salary + citizen_acc + support # --- القسم الثاني: المصاريف --- st.header("💸 المصاريف والالتزامات") # قائمة المصاريف الافتراضية default_expenses = { "بر الوالدة": 300, "قسط البيت": 6566, "الكهرباء": 744, "الإنترنت": 690, "الماء": 100, "زبود": 1600, "نقل طيف ولمى": 350, "قرض": 2950, "تجميع الديرة": 2500 } # تمكين المستخدم من تعديل القيم أو إضافة خانات updated_expenses = {} for item, value in default_expenses.items(): updated_expenses[item] = st.sidebar.number_input(f"تعديل {item}", value=value) # إضافة خانة جديدة ديناميكياً new_item = st.sidebar.text_input("إضافة مصروف جديد (الاسم)") new_value = st.sidebar.number_input("قيمة المصروف الجديد", value=0) if new_item: updated_expenses[new_item] = new_value total_expenses = sum(updated_expenses.values()) remaining = total_income - total_expenses # --- القسم الثالث: العرض والنتائج --- st.divider() c1, c2, c3 = st.columns(3) c1.metric("إجمالي الدخل", f"{total_income} ريال") c2.metric("إجمالي المصاريف", f"{total_expenses} ريال") c3.metric("المتبقي الصافي", f"{remaining} ريال", delta_color="normal") # عرض جدول المصاريف st.subheader("📊 تفاصيل المصاريف") df = pd.DataFrame(list(updated_expenses.items()), columns=['البند', 'المبلغ']) st.table(df) if st.button("حفظ التقرير نهائياً"): st.success("تم الحساب بنجاح! يمكنك تصوير الشاشة للتوثيق.")
